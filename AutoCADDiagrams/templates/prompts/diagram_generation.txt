"Show how a user interacts with Copilot Studio to query data from SharePoint and Dataverse"

## Example Response

```python
from diagrams import Diagram, Cluster, Edge
from diagrams.onprem.client import User
from m365_diagrams.icons import CopilotStudio, SharePoint, Dataverse, GraphAPI

def generate_diagram(output_path: str, filename: str = "copilot_data_query"):
    with Diagram(
        "Copilot Studio Data Query Flow",
        filename=f"{output_path}/{filename}",
        show=False,
        direction="LR"
    ):
        user = User("User")

        with Cluster("Copilot Studio"):
            agent = CopilotStudio("Custom Agent")

        with Cluster("Data Sources"):
            graph = GraphAPI("Graph API")
            sp = SharePoint("SharePoint")
            dv = Dataverse("Dataverse")

        user >> Edge(label="asks question") >> agent
        agent >> Edge(label="queries") >> graph
        graph >> sp
        graph >> dv
        agent >> Edge(label="returns answer", style="dashed") >> user
```

- ALWAYS import User from `diagrams.onprem.client`
